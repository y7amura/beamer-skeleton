% skip page numbering
\newcommand{\skipnumbering}{\addtocounter{framenumber}{-1}}

% appendix
\newcommand{\beginappendix}{%
  {%
    \setbeamercolor{background canvas}{bg=black}
    \begin{frame}<handout:0>[plain]
    \end{frame}
  }
  \skipnumbering%
  \appendix%
}

% paragraph
\newlength{\pskiplength}
\setlength{\pskiplength}{4pt}
\newcommand{\pskip}{\vspace{\pskiplength}\par}
\newenvironment{paragraph}{}{\pskip}

% indent block (maybe not required anymore)
\newenvironment{indentblock}{
  \begin{columns}
  \begin{column}{0.05\textwidth}
  \end{column}
  \begin{column}{0.95\textwidth}
}{
  \end{column}
  \end{columns}
}
